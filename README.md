<h1>Commands to run after cloning</h1>
<ol>
    <li>composer install</li>
    <li>npm install</li>
    <li>npm run dev</li>
    <li>php artisan migrate</li>
    <li>composer dump-autoload</li>
    <li>php artisan db:seed</li>
</ol>

<h1>To open the server</h1>
<ul>
    <li>php artisan serve</li>
</ul>
